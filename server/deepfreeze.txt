app.get('/api/leaderboard', async (req, res) => {
    const db = req.app.get('db')

    let leaders = db.leaderboard()

    res.status(200).send(leaders)
})


let showTurn = arr => {
	let moves = []
	for (i=0; i <arr.length; i++) {
		for (key in arr[i]) {
			if (key == 'san') {
				let something = Object.values(arr[i].san).join('')
				moves.push(something)
			}
		} 
	}
	return moves
}
